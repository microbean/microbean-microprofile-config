065/**
066 * An {@link Extension} that enables injection of {@link
067 * ConfigProperty}-annotated configuration property values.
068 *
069 * @author <a href="https://about.me/lairdnelson"
070 * target="_parent">Laird Nelson</a>
071 */
072public final class ConfigExtension implements Extension {
073
074  private static final Map<Class<?>, Class<?>> wrapperClasses;
075
076  static {
077    wrapperClasses = new HashMap<>();
078    wrapperClasses.put(boolean.class, Boolean.class);
079    wrapperClasses.put(byte.class, Byte.class);
080    wrapperClasses.put(char.class, Character.class);
081    wrapperClasses.put(double.class, Double.class);
082    wrapperClasses.put(float.class, Float.class);
083    wrapperClasses.put(int.class, Integer.class);
084    wrapperClasses.put(long.class, Long.class);
085    wrapperClasses.put(short.class, Short.class);
086  }
087
088  private final Map<Set<Annotation>, Set<InjectionPoint>> configPropertyInjectionPoints;
089
090  private final Set<Set<Annotation>> allConfigQualifiers;
091  
092  /**
093   * Creates a new {@link ConfigExtension}.
094   */
095  public ConfigExtension() {
096    super();
097    this.configPropertyInjectionPoints = new HashMap<>();
098    this.allConfigQualifiers = new HashSet<>();
099  }
100
101  private final <X> void processBean(@Observes final ProcessBean<X> event) {
102    if (event != null) {
103      final Bean<X> bean = event.getBean();
104      if (bean != null) {
105        final Set<InjectionPoint> beanInjectionPoints = bean.getInjectionPoints();
106        if (beanInjectionPoints != null && !beanInjectionPoints.isEmpty()) {
107          for (final InjectionPoint beanInjectionPoint : beanInjectionPoints) {
108            if (beanInjectionPoint != null) {
109              final Set<Annotation> qualifiers = beanInjectionPoint.getQualifiers();
110              final Type type = beanInjectionPoint.getType();
111              if (Config.class.equals(type)) {
112                if (qualifiers != null && !qualifiers.isEmpty()) {
113                  final Set<Annotation> configQualifiers = new HashSet<>(qualifiers);
114                  configQualifiers.add(AnyLiteral.INSTANCE);
115                  allConfigQualifiers.add(configQualifiers);
116                }
117              } else {
118                final Annotated annotated = beanInjectionPoint.getAnnotated();
119                if (annotated != null && annotated.isAnnotationPresent(ConfigProperty.class)) {
120                  Set<InjectionPoint> configPropertyInjectionPoints = this.configPropertyInjectionPoints.get(qualifiers);
121                  if (configPropertyInjectionPoints == null) {
122                    configPropertyInjectionPoints = new HashSet<>();
123                    this.configPropertyInjectionPoints.put(qualifiers, configPropertyInjectionPoints);
124                  }
125                  configPropertyInjectionPoints.add(beanInjectionPoint);
126                }
127              }
128            }
129          }
130        }
131      }
132    }
133  }
134
135  private final <T, X> void processObserverMethod(@Observes final ProcessObserverMethod<T, X> event,
136                                                  final BeanManager beanManager) {
137    if (event != null && beanManager != null) {
138      final ObserverMethod<T> observerMethod = event.getObserverMethod();
139      if (observerMethod != null) {
140        final AnnotatedMethod<X> annotatedMethod = event.getAnnotatedMethod();
141        if (annotatedMethod != null) {
142          final List<AnnotatedParameter<X>> annotatedParameters = annotatedMethod.getParameters();
143          if (annotatedParameters != null && annotatedParameters.size() > 1) {
144            for (final AnnotatedParameter<X> annotatedParameter : annotatedParameters) {
145              if (annotatedParameter != null &&
146                  !annotatedParameter.isAnnotationPresent(Observes.class)) {
147                final InjectionPoint injectionPoint = beanManager.createInjectionPoint(annotatedParameter);
148                assert injectionPoint != null;
149                final Set<Annotation> qualifiers = injectionPoint.getQualifiers();
150                if (Config.class.equals(injectionPoint.getType())) {
151                  if (qualifiers != null && !qualifiers.isEmpty()) {
152                    final Set<Annotation> configQualifiers = new HashSet<>(qualifiers);
153                    configQualifiers.add(AnyLiteral.INSTANCE);
154                    allConfigQualifiers.add(configQualifiers);
155                  }
156                } else if (annotatedParameter.isAnnotationPresent(ConfigProperty.class)) {
157                  Set<InjectionPoint> configPropertyInjectionPoints = this.configPropertyInjectionPoints.get(qualifiers);
158                  if (configPropertyInjectionPoints == null) {
159                    configPropertyInjectionPoints = new HashSet<>();
160                    this.configPropertyInjectionPoints.put(qualifiers, configPropertyInjectionPoints);
161                  }
162                  configPropertyInjectionPoints.add(injectionPoint);
163                }
164              }
165            }
166          }
167        }
168      }
169    }
170  }

172  private final void afterBeanDiscovery(@Observes final AfterBeanDiscovery event,
173                                        final BeanManager beanManager) {
174    if (event != null) {
175
176      final AnnotatedType<Config> configAnnotatedType = beanManager.createAnnotatedType(Config.class);
177      assert configAnnotatedType != null;
178
179      final Set<Annotation> defaultConfigQualifiers = new HashSet<>();
180      defaultConfigQualifiers.add(AnyLiteral.INSTANCE);
181      defaultConfigQualifiers.add(DefaultLiteral.INSTANCE);
182      this.allConfigQualifiers.add(defaultConfigQualifiers);
183      
184      if (!this.configPropertyInjectionPoints.isEmpty()) {
185
186        final Set<Type> types = new HashSet<>();
187        final Set<Entry<Set<Annotation>, Set<InjectionPoint>>> entrySet = this.configPropertyInjectionPoints.entrySet();
188        assert entrySet != null;
189        assert !entrySet.isEmpty();
190        for (final Entry<Set<Annotation>, Set<InjectionPoint>> entry : entrySet) {
191          assert entry != null;
192          final Set<Annotation> qualifiers = entry.getKey();
193          final Set<InjectionPoint> configPropertyInjectionPoints = entry.getValue();
194          assert configPropertyInjectionPoints != null;
195          for (final InjectionPoint injectionPoint : configPropertyInjectionPoints) {
196            assert injectionPoint != null;
197            Type type = injectionPoint.getType();
198            assert type != null;
199            if (type instanceof Class && ((Class<?>)type).isPrimitive()) {
200              type = wrapperClasses.get(type);
201              assert type != null;
202            }
203            if (types.add(type) && noBeans(beanManager, type, qualifiers)) {
204              event.addBean(new ConfigPropertyBean<>(type, qualifiers));
205            }
206          }
207        }
208        types.clear();
209      }
210
211      for (final Set<Annotation> configQualifiers : this.allConfigQualifiers) {
212        if (noBeans(beanManager, Config.class, configQualifiers)) {
213          event.addBean(new ConfigBean(configQualifiers));
214        }
215      }
216
217    }
218    this.allConfigQualifiers.clear();
219  }

221  private final void afterDeploymentValidation(@Observes final AfterDeploymentValidation event,
222                                               final BeanManager beanManager) {
223    if (event != null && beanManager != null) {
224      final CreationalContext<?> cc = beanManager.createCreationalContext(null);
225      try {
226        final Set<Entry<Set<Annotation>, Set<InjectionPoint>>> entrySet = this.configPropertyInjectionPoints.entrySet();
227        assert entrySet != null;
228        for (final Entry<Set<Annotation>, Set<InjectionPoint>> entry : entrySet) {
229          assert entry != null;
230          final Set<Annotation> qualifiers = entry.getKey();
231          final Set<InjectionPoint> configPropertyInjectionPoints = entry.getValue();
232          assert configPropertyInjectionPoints != null;
233          for (final InjectionPoint injectionPoint : configPropertyInjectionPoints) {
234            assert injectionPoint != null;
235            if (beanManager.getInjectableReference(injectionPoint, cc) == null) {
236              event.addDeploymentProblem(new DeploymentException("No value exists for the mandatory configuration property named " +
237                                                                 getConfigPropertyName(injectionPoint)));
238            }
239          }
240        }
241      } finally {
242        cc.release();
243      }
244    }
245    this.configPropertyInjectionPoints.clear();
246  }

248  private static final void onShutdown(@Observes @Destroyed(ApplicationScoped.class) final Object event) throws Exception {
249    final ConfigProviderResolver configProviderResolver = ConfigProviderResolver.instance();
250    if (configProviderResolver instanceof AutoCloseable) {
251      ((AutoCloseable)configProviderResolver).close();
252    }
253  }
254
255  private static final boolean noBeans(final BeanManager beanManager, final Type type, final Set<Annotation> qualifiers) {
256    Objects.requireNonNull(beanManager);
257    Objects.requireNonNull(type);
258    final Collection<?> beans;
259    if (qualifiers == null || qualifiers.isEmpty()) {
260      beans = beanManager.getBeans(type);
261    } else {
262      beans = beanManager.getBeans(type, qualifiers.toArray(new Annotation[qualifiers.size()]));
263    }
264    return beans == null || beans.isEmpty();
265  }
266
267  static final String getConfigPropertyName(final InjectionPoint injectionPoint) {
268    return getConfigPropertyName(injectionPoint, null);
269  }
270
271  static final String getConfigPropertyName(final InjectionPoint injectionPoint, ConfigProperty configProperty) {
272    String name = null;
273    if (injectionPoint != null) {
274      if (configProperty == null) {
275        final Collection<? extends Annotation> qualifiers = injectionPoint.getQualifiers();
276        if (qualifiers != null) {
277          for (final Annotation qualifier : qualifiers) {
278            if (qualifier != null && ConfigProperty.class.equals(qualifier.annotationType())) {
279              configProperty = (ConfigProperty)qualifier;
280              break;
281            }
282          }
283        }
284      }
285      if (configProperty != null) {
286        name = configProperty.name();
287        assert name != null;
288        if (name.isEmpty()) {
289          final Annotated annotated = injectionPoint.getAnnotated();
290          if (annotated instanceof AnnotatedField) {
291            final AnnotatedField<?> field = (AnnotatedField<?>)annotated;
292            final Member member = field.getJavaMember();
293            if (member != null) {
294              final AnnotatedType<?> declaringType = field.getDeclaringType();
295              if (declaringType != null) {
296                name = declaringType.getJavaClass().getCanonicalName() + "." + member.getName();
297              }
298            }
299          } else if (annotated instanceof AnnotatedParameter) {
300            final AnnotatedParameter<?> annotatedParameter = (AnnotatedParameter<?>)annotated;
301            // The specification says: "[The ConfigProperty name that
302            // will be synthesized if one is not provided] will be
303            // derived automatically as
304            // <class_name>.<injetion_point_name> [sic], where
305            // injection_point_name [sic] is the field name or
306            // parameter name, class_name is the fully qualified name
307            // of the class being injected to [sic]."
308            final AnnotatedCallable<?> declaringCallable = annotatedParameter.getDeclaringCallable();
309            if (declaringCallable != null) {
310              final Member member = declaringCallable.getJavaMember();
311              assert member instanceof Executable;
312              final Executable executable = (Executable)member;
313              final int position = annotatedParameter.getPosition();
314              final Parameter[] parameters = executable.getParameters();
315              if (parameters != null && parameters.length > position) {
316                final Parameter parameter = parameters[position];
317                assert parameter != null;
318                if (parameter.isNamePresent()) {
319                  final AnnotatedType<?> declaringType = declaringCallable.getDeclaringType();
320                  if (declaringType != null) {
321                    name = declaringType.getJavaClass().getCanonicalName() + "." + parameter.getName();
322                  }
323                }
324              }
325            }
326          } else {
327            assert false;
328          }
329        }
330      }
331    }
332    return name;
333  }
